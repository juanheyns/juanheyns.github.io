  <script>
    const pool = ["A", "E", "T", "R", "S", "N", "L", "O", "I", "D", "G", "P", "Q", "Z", "U", "M"];
    const dictionary = new Set(["SNAKE", "SNAKES", "WORD", "WORDS", "GRID", "MOVE", "TAP", "TAPS", "SCORE", "FAST"]);
    const reserved = [];

    const letterPoolEl = document.getElementById("letterPool");
    const currentWordEl = document.getElementById("currentWord");
    const reservedRowEl = document.getElementById("reservedRow");
    const validationEl = document.getElementById("validation");

    function renderPool() {
      letterPoolEl.innerHTML = "";
      pool.forEach((letter, index) => {
        const tile = document.createElement("button");
        tile.className = "tile";
        tile.textContent = letter;
        tile.addEventListener("click", () => reserveLetter(letter, index, tile));
        letterPoolEl.appendChild(tile);
      });
    }

    function reserveLetter(letter, index, tile) {
      const reservation = {
        letter,
        id: Date.now() + Math.random(),
        index,
      };
      reserved.push(reservation);
      tile.classList.add("reserved");
      tile.disabled = true;

      updateWord();
      window.setTimeout(() => releaseLetter(reservation), 3000);
    }

    function releaseLetter(reservation) {
      const idx = reserved.findIndex((item) => item.id === reservation.id);
      if (idx === -1) return;
      reserved.splice(idx, 1);
      const tile = letterPoolEl.children[reservation.index];
      if (tile) {
        tile.classList.remove("reserved");
        tile.disabled = false;
      }
      updateWord();
    }

    function updateWord() {
      const word = reserved.map((item) => item.letter).join("");
      currentWordEl.textContent = word || "—";
      reservedRowEl.innerHTML = "";
      reserved.forEach((item) => {
        const chip = document.createElement("span");
        chip.className = "chip";
        chip.textContent = `${item.letter} · 3s`;
        reservedRowEl.appendChild(chip);
      });
      if (!word) {
        validationEl.textContent = "Tap letters to reserve. Valid words let you move.";
        return;
      }
      validationEl.textContent = dictionary.has(word)
        ? `✅ Valid word (${word.length} letters). Swipe to move!`
        : "❌ Invalid word. Keep tapping letters.";
    }

    function renderGrid() {
      const grid = document.getElementById("grid");
      grid.innerHTML = "";
      for (let row = 0; row < 15; row += 1) {
        for (let col = 0; col < 15; col += 1) {
          const cell = document.createElement("div");
          cell.className = "cell";
          if (row === 7 && col === 7) {
            cell.classList.add("head");
            cell.textContent = "P1";
          } else if (row === 7 && col < 7 && col > 2) {
            cell.classList.add("body");
            cell.textContent = pool[col % pool.length];
          } else if (row === 4 && col === 10) {
            cell.classList.add("neutral");
            cell.textContent = "✖";
          }
          grid.appendChild(cell);
        }
      }
    }

    document.querySelectorAll(".direction").forEach((button) => {
      button.addEventListener("click", () => {
        const dir = button.dataset.dir;
        const word = reserved.map((item) => item.letter).join("");
        if (!dictionary.has(word)) {
          validationEl.textContent = "❌ Invalid word. You cannot move yet.";
          return;
        }
        validationEl.textContent = `Snake moves ${dir}. Letters placed!`;
      });
    });

    renderGrid();
    renderPool();
  </script>
