      <div class="letter-pool" id="letterPool"></div>

      <div class="current-word" id="currentWord"></div>
      <div class="actions">
        <button class="primary" id="confirmWord">Confirm Word</button>
        <button id="clearWord">Clear</button>
        <button class="danger" id="cancelMove">Cancel Move</button>
      </div>
      <p class="tip">Rare letters grant a speed boost for one move only.</p>

  <script>
    const arena = document.getElementById("arena");
    const letterPool = document.getElementById("letterPool");
    const currentWord = document.getElementById("currentWord");
    const confirmWord = document.getElementById("confirmWord");
    const clearWord = document.getElementById("clearWord");
    const cancelMove = document.getElementById("cancelMove");

    const poolLetters = ["A", "R", "S", "T", "E", "L", "N", "O", "Q", "Z", "M", "D", "U", "I", "P", "G"];
    const selectedLetters = [];

    const tileState = new Map([
      ["7-7", { type: "head", text: "H" }],
      ["7-6", { type: "body", text: "E" }],
      ["7-5", { type: "body", text: "L" }],
      ["7-4", { type: "body", text: "L" }],
      ["7-3", { type: "body", text: "O" }],
      ["10-9", { type: "head", text: "B" }],
      ["10-8", { type: "body", text: "O" }],
      ["10-7", { type: "body", text: "T" }],
      ["4-10", { type: "neutral", text: "#" }],
      ["5-10", { type: "neutral", text: "#" }]
    ]);

    const renderArena = () => {
      arena.innerHTML = "";
      for (let row = 0; row < 15; row += 1) {
        for (let col = 0; col < 15; col += 1) {
          const key = `${row}-${col}`;
          const tile = document.createElement("div");
          tile.className = "tile";
          const data = tileState.get(key);
          if (data) {
            tile.classList.add(data.type);
            tile.innerHTML = `<span>${data.text}</span>`;
          }
          arena.appendChild(tile);
        }
      }
    };

    const renderPool = () => {
      letterPool.innerHTML = "";
      poolLetters.forEach((letter, index) => {
        const button = document.createElement("button");
        button.type = "button";
        button.className = "letter";
        button.textContent = letter;
        button.dataset.index = index;
        if (selectedLetters.includes(index)) {
          button.classList.add("selected");
        }
        button.addEventListener("click", () => {
          if (selectedLetters.includes(index)) {
            selectedLetters.splice(selectedLetters.indexOf(index), 1);
          } else {
            selectedLetters.push(index);
          }
          updateWord();
          renderPool();
        });
        letterPool.appendChild(button);
      });
    };

    const updateWord = () => {
      currentWord.textContent = selectedLetters.map((index) => poolLetters[index]).join("");
    };

    confirmWord.addEventListener("click", () => {
      if (!selectedLetters.length) {
        currentWord.textContent = "";
        return;
      }
      currentWord.textContent = `${currentWord.textContent} ✓ Valid`;
    });

    clearWord.addEventListener("click", () => {
      selectedLetters.length = 0;
      updateWord();
      renderPool();
    });

    cancelMove.addEventListener("click", () => {
      selectedLetters.length = 0;
      updateWord();
      renderPool();
    });

    renderArena();
    renderPool();
    updateWord();
  </script>
